#!/usr/bin/python

import RPi.GPIO as GPIO

class Sensor:
  def __init__(self, *pins):
    self.__mappings = {}
    self.__setup(pins)
    GPIO.setwarnings(False)
    GPIO.setmode(GPIO.BOARD)
    
  def __setup(self, pins):
    self.__mappings = {}
    team = 0
    for pin in pins:
      team += 1
      GPIO.setup(pin, GPIO.IN)
      GPIO.add_event_detect(pin, GPIO.FALLING, callback=self.__teamscores)
      self.__mappings[pin] = "Team %d" % team

  def __teamscores(self, channel):
    print "Team: %s scores on channel: %s" % self.mappings[channel] channel
    # print "CHANNEL: %s" % channel
